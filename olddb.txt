model User {
  userID       Int           @id @default(autoincrement())
  email        String        @unique
  firstname    String
  lastname     String
  username     String        @unique
  password     String
  transactions Transaction[] @relation("UserTransactionsUser")
  watchlist    Watchlist[]   @relation("UserWatchlistUser")
}

model Transaction {
  transactionID Int      @id @default(autoincrement())
  amount        Float
  price         Float
  date          DateTime
  userID        Int
  user          User     @relation(fields: [userID], references: [userID], name: "UserTransactionsUser")
}

model Watchlist {
  watchlistID Int    @id @default(autoincrement())
  coinID      String
  userID      Int
  user        User   @relation(fields: [userID], references: [userID], name: "UserWatchlistUser")
}
